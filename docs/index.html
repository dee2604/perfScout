Quick Start

1. Add Dependency

dependencies { implementation("com.github.deekshasinghal326:perfScout:1.1.3") }

2. Basic Usage

// Get performance metrics val cpuInfo = PerfScout.getCpuInfo() val ramInfo = PerfScout.getRamInfo(context) val batteryInfo = PerfScout.getBatteryInfo(context) // Monitor crashes and ANRs PerfScout.setCrashListener { crashInfo -> Log.e("PerfScout", "Crash detected: $crashInfo") } PerfScout.setAnrListener { anrInfo -> Log.w("PerfScout", "ANR detected: $anrInfo") }

3. Advanced Features

// Frame rendering analysis PerfScout.getFrameRenderingInfo(3000) { frameInfo -> println("Performance Score: ${frameInfo.performanceScore}") println("Jank Percentage: ${frameInfo.jankPercentage}%") } // Memory leak monitoring PerfScout.monitorMemoryLeaks(context) { isLeaking, usageHistory -> if (isLeaking) Log.w("PerfScout", "Possible memory leak: $usageHistory") }
