Quick Start

1. Add Dependency

dependencies { implementation("com.github.dee2604:perfScout:1.1.8") }

2. Basic Usage

// Get performance metrics (new delegate-based API) val cpuResult = PerfScoutMetrics.cpu.get() val ramResult = PerfScoutMetrics.ram.get(context) val batteryResult = PerfScoutMetrics.battery.get(context) // Handle results safely when (cpuResult) { is PerfResult.Success -> Log.d("PerfScout", "CPU: ${cpuResult.info}") is PerfResult.Error -> Log.e("PerfScout", "Error: ${cpuResult.message}") } // Monitor crashes and ANRs PerfScoutMetrics.setCrashListener { crashInfo -> Log.e("PerfScout", "Crash detected: $crashInfo") } PerfScoutMetrics.setAnrListener { anrInfo -> Log.w("PerfScout", "ANR detected: $anrInfo") }

3. Advanced Features

// Frame rendering analysis (async recommended) lifecycleScope.launch { val frameResult = PerfScoutMetrics.frameRendering.getAsync() when (frameResult) { is PerfResult.Success -> { println("Performance Score: ${frameResult.info.performanceScore}") println("Jank Percentage: ${frameResult.info.jankPercentage}%") } is PerfResult.Error -> Log.e("PerfScout", "Error: ${frameResult.message}") } } // Media quality analysis val analysisResult = PerfScoutMetrics.mediaQualityAnalysis.get(context) when (analysisResult) { is PerfResult.Success -> { println("Network Speed: ${analysisResult.info.networkSpeed}") println("Recommended Quality: ${analysisResult.info.recommendedQuality}") } is PerfResult.Error -> Log.e("PerfScout", "Error: ${analysisResult.message}") }
